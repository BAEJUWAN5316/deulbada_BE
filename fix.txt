server {
    listen 80;
    server_name 43.201.70.73; # 서버 IP 주소

    client_max_body_size 5M; # 파일 업로드 용량 제한 (이전에 설정했던 값)

    # 정적 파일 처리
    location /static/ {
        alias /home/ubuntu/deulbada_BE/staticfiles/;
    }

    # 미디어 파일 처리
    location /media/ {
        alias /home/ubuntu/deulbada_BE/media/;
    }

    # 🌐 WebSocket 요청은 Daphne으로 전달
    location /ws/ {
        proxy_pass http://127.0.0.1:8001; # Daphne 포트
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # 일반 HTTP 요청은 Gunicorn으로 전달
    location / {
        proxy_pass http://127.0.0.1:8000; # Gunicorn 포트
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
